/* ─────────────────────────────────────
   ACCORDION
───────────────────────────────────── */
function toggleAccordion(trigger) {
  const body = trigger.nextElementSibling;
  const isOpen = trigger.classList.contains('open');
  document.querySelectorAll('.accordion-trigger').forEach(t => {
    t.classList.remove('open');
    t.nextElementSibling.classList.remove('open');
  });
  if (!isOpen) { trigger.classList.add('open'); body.classList.add('open'); }
}

let currentStep = 1;
const totalSteps = 4;
const progressWidths = { 1:'25%', 2:'50%', 3:'75%', 4:'100%' };
const labelIds = ['pl-1','pl-2','pl-3','pl-4'];

function updateProgress(step) {
  document.getElementById('progressFill').style.width = progressWidths[step];
  document.getElementById('stepText').textContent = `Step ${step} of ${totalSteps}`;
  labelIds.forEach((id, i) =>
    document.getElementById(id).classList.toggle('active', i + 1 === step));
}
function showStep(step) {
  document.querySelectorAll('.form-step').forEach(s => s.classList.remove('active'));
  document.getElementById('step-' + step).classList.add('active');
  updateProgress(step);
  document.getElementById('apply').scrollIntoView({ behavior: 'smooth', block: 'start' });
}

/* Validation */
function setValid(id)   { const f=document.getElementById(id); f.classList.remove('error'); f.classList.add('valid'); }
function setError(id)   { const f=document.getElementById(id); f.classList.add('error'); f.classList.remove('valid'); }
function clearState(id) { const f=document.getElementById(id); f.classList.remove('error','valid'); }

function liveValidateName()  { document.getElementById('inp-name').value.trim().length>1?setValid('field-name'):clearState('field-name'); }
function liveValidateAge()   { const v=parseInt(document.getElementById('inp-age').value); (!isNaN(v)&&v>=16&&v<100)?setValid('field-age'):clearState('field-age'); }
function liveValidateEmail() { /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(document.getElementById('inp-email').value)?setValid('field-email'):clearState('field-email'); }

function formatPhone(input) {
  let v = input.value.replace(/\D/g,'');
  if (v.length<=3) input.value=v;
  else if (v.length<=6) input.value=v.slice(0,3)+' '+v.slice(3);
  else input.value=v.slice(0,3)+' '+v.slice(3,6)+' '+v.slice(6,10);
}

function validateStep(step) {
  let ok = true;
  if (step===1) {
    const name=document.getElementById('inp-name').value.trim();
    const age=parseInt(document.getElementById('inp-age').value);
    if (name.length<2){setError('field-name');ok=false;}else setValid('field-name');
    if (isNaN(age)||age<16||age>=100){setError('field-age');ok=false;}else setValid('field-age');
  }
  if (step===2) {
    const email=document.getElementById('inp-email').value;
    const country=document.getElementById('inp-country').value;
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)){setError('field-email');ok=false;}else setValid('field-email');
    if (!country){setError('field-country');ok=false;}else setValid('field-country');
  }
  if (step===3) {
    const interest=document.getElementById('inp-interest').value.trim();
    const goals=document.getElementById('inp-goals').value.trim();
    if (interest.length<10){setError('field-interest');ok=false;}else setValid('field-interest');
    if (goals.length<10){setError('field-goals');ok=false;}else setValid('field-goals');
  }
  if (step===4) {
    const situation=document.getElementById('inp-situation').value.trim();
    const why=document.getElementById('inp-why').value.trim();
    if (situation.length<10){setError('field-situation');ok=false;}else setValid('field-situation');
    if (why.length<10){setError('field-why');ok=false;}else setValid('field-why');
  }
  return ok;
}

function nextStep(from) {
  if (!validateStep(from)) return;
  currentStep = from+1;
  showStep(currentStep);
}
function prevStep(from) {
  currentStep = from-1;
  showStep(currentStep);
}

/* ─────────────────────────────────────
   FORM SUBMIT — sends to Formspree
   IMPORTANT: Replace YOUR_FORM_ID below
   with your actual Formspree form ID.
   Sign up free at https://formspree.io
───────────────────────────────────── */
async function submitForm() {
  if (!validateStep(4)) return;

  const btn = document.getElementById('submitBtn');
  btn.textContent = 'Submitting...';
  btn.disabled = true;

  const payload = {
    name:      document.getElementById('inp-name').value.trim(),
    age:       document.getElementById('inp-age').value,
    phone:     document.getElementById('inp-phone').value,
    email:     document.getElementById('inp-email').value,
    country:   document.getElementById('inp-country').value,
    interest:  document.getElementById('inp-interest').value.trim(),
    goals:     document.getElementById('inp-goals').value.trim(),
    situation: document.getElementById('inp-situation').value.trim(),
    heard:     document.getElementById('inp-heard').value.trim(),
    why:       document.getElementById('inp-why').value.trim(),
  };

  /* ──────────────────────────────────────
     TO MAKE THIS LIVE:
     1. Go to https://formspree.io and create a free account
     2. Create a new form — Formspree gives you an endpoint like:
        https://formspree.io/f/xyzabcde
     3. Replace the URL below with your endpoint
     4. Formspree will email you every submission
     ────────────────────────────────────── */
  const FORMSPREE_URL = 'https://formspree.io/f/YOUR_FORM_ID';

  try {
    const res = await fetch(FORMSPREE_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
      body: JSON.stringify(payload)
    });

    if (res.ok) {
      // Success state
      document.querySelectorAll('.form-step').forEach(s => s.classList.remove('active'));
      document.getElementById('step-success').classList.add('active');
      document.getElementById('progressFill').style.width = '100%';
      document.getElementById('stepText').textContent = 'Application complete';
      labelIds.forEach(id => document.getElementById(id).classList.remove('active'));
    } else {
      throw new Error('Submit failed');
    }
  } catch (err) {
    // Fallback: show success UI anyway (for demo/testing)
    // Remove the lines below and replace with an error message
    // once you have a real Formspree endpoint connected.
    document.querySelectorAll('.form-step').forEach(s => s.classList.remove('active'));
    document.getElementById('step-success').classList.add('active');
    document.getElementById('progressFill').style.width = '100%';
    document.getElementById('stepText').textContent = 'Application complete';
    labelIds.forEach(id => document.getElementById(id).classList.remove('active'));
  }
}
